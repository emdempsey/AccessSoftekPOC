$prefix = "mac"

$vnetAddressSpacePrefix = "10.0"
$sqlServerServiceAccountPassword = "Office365"
$sqlServerServiceAccountPassword = ConvertTo-SecureString -AsPlainText -Force -String $sqlServerServiceAccountPassword
$sqlServerServiceAccountUsername = "sqlService"

***Use this command for AzureDeploy.json***
New-AzureResourceGroup -Name $ResourceGroupName -Location $location -TemplateFile $TemplateFile -customerPrefix $prefix -dnsNameForPublicIP $dnsname -Force -Verbose

***Use this command for templates with Default values***
New-AzureResourceGroup -Name $ResourceGroupName -Location $location -TemplateFile $TemplateFile -customerPrefix $prefix -VirtualNetworkName $vnetname -Force -Verbose

***Use this command for environment child templates *EnvApp, *EnvData, *EnvWeb***
New-AzureResourceGroup -Name $ResourceGroupName -Location $location -TemplateFile $TemplateFile -adminUsername $adminusername -adminPassword $adminPassword -customerLocation $location -customerPrefix $Prefix -DomainName $DomainName -VirtualNetworkName $vnetname -Force -Verbose


***Use this command to deploy the SQL PaaS***
New-AzureResourceGroup -Name $ResourceGroupName -Location $location -TemplateFile $TemplateFile -customerLocation $location -customerPrefix $prefix -sqlServerServiceAccountPassword $sqlServerServiceAccountPassword -sqlServerServiceAccountUsername $sqlServerServiceAccountUsername -vnetAddressSpacePrefix $vnetAddressSpacePrefix -Force -Verbose
